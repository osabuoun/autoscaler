JDL : JOB_DEADLINE (in seconds)

TC	: TASK_COUNT
JDN : JOB_DUATION_NEEDED = TASK_COUNT * TASK_DURATION
SIN : SIMULTANEOUS_INSTANCES_NEEDED = JOB_DEADLINE / JOB_DUATION_NEEDED
